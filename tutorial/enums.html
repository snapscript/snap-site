Types

enum A {
   ONE(1),
   TWO(2),
   THREE(3);

   const x;
   
   new(x) {
      this.x = x;
   }
}

class B {
   const x;

   new(x) {
      this.x = x;
   }
}

trait C {
   func();
}

class D extends B with C {
   const y;

   new(x, y) : super(x) {
      this.y = y;
   }

   override func() {
      return x+y;
   }
}

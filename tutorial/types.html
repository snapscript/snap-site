Types

trait A {
   foo();
   bar() {
      return foo() + 2;
   }
}

class B {
   const x;

   new(x) {
      this.x = x;
   }
}

class C extends B with A {
   const y;

   new(x, y) : super(x) {
      this.y = y;
   }

   override foo() {
      return x+y;
   }

   override hashCode(){
      return x ^ y;
   }

   override equals(a) {
      return a.x == x && a.y == y;
   }
}
